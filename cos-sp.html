 void scheduler::fcfs()
 {
     average_wait_time = 0;
     pool::eval_prcs_prty();
     auto pit = pool::ready_queue.begin();
     while (pit != pool::ready_queue.end())
     {
        exec(pit, pit->get_ttl());

        dispatcher::context_switch(pit, pit->get_ttl());

        PSAscreen::get().draw_process_exec(average_wait_time,
                                           *pit, pool::done_queue);
     }
 }
